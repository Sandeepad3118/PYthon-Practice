```#two pointer approach - O(n)

def sec_lar(arr):
    largest = arr[0]
    seclar = -1
    for i in range(1, len(arr)):
        if arr[i]>largest:
            seclar = largest
            largest = arr[i]
        elif (arr[i]<largest and arr[i]>seclar):
            seclar = arr[i]
        else:
            continue
    print(seclar)  
    print(largest)    


arr= [3,1,5,2]
sec_lar(arr)```
